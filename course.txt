    $(document).ready(function () {

      const accordions = document.querySelectorAll('.accordion');

      accordions.forEach(accordion => {
        const header = accordion.querySelector('.accordion-header');
        const content = accordion.querySelector('.accordion-content');

        header.addEventListener('click', () => {
          // Load the iframe only when the content becomes active
          if (!content.classList.contains('active')) {
            const iframe = content.querySelector('iframe');
            if (iframe.dataset.src) {
              iframe.src = iframe.dataset.src;
              delete iframe.dataset.src; // Remove the data-src attribute to avoid reloading the iframe
            }
          }

          // Toggle the 'active' class on the accordion content to show/hide it
          header.classList.toggle('active');
          content.classList.toggle('active');

          // If the content is active, close all other accordion contents
          if (content.classList.contains('active')) {
            accordions.forEach(item => {
              if (item !== accordion) {
                item.querySelector('.accordion-header').classList.remove('active');
                item.querySelector('.accordion-content').classList.remove('active');
              }
            });
          }
        });
      });
    });
